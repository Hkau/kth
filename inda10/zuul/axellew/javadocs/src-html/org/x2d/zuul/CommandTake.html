001    package org.x2d.zuul;
002    import java.util.*;
003    /**
004     * Command word take. This command is used when the player is trying to take an item and put it into
005     * the backpack.
006     */
007    public class CommandTake extends CommandWord {
008        public CommandTake(String commandName) {
009            super(commandName);
010        }
011        
012        @Override
013        public String[] getTargets(Game game) {
014            Collection<Items> rItems = game.getPlayer().getCurrentRoom().getItems();
015            ArrayList<String> items = new ArrayList<String>();
016            for (Items i : rItems) {
017                if (i.getItem().isTakable()) {
018                    items.add(i.getItem().getName());
019                }
020            }
021            return items.toArray(new String[0]);
022        }
023        
024        /**
025         * @param target The name of the item which the player should try to take.
026         */
027        @Override
028        public void executeCommand(Game game, String target) {
029            Items item = Items.getItem(target);
030            if (!game.getPlayer().getCurrentRoom().hasItem(item)) {
031                System.out.println(String.format("Det finns inget föremål med namnet '%s' i det här rummet.", target));
032                return;
033            }
034            if (!item.getItem().isTakable()) {
035                System.out.println("Du kan inte ta med dig: "+target);
036                return;
037            }
038            game.getPlayer().getCurrentRoom().removeItem(item);
039            game.getPlayer().addItem(item);
040        }
041        @Override
042        public boolean mustHaveTarget() {
043            return true;
044        }
045    
046    }
