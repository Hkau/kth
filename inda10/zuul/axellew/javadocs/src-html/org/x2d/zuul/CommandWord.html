001    package org.x2d.zuul;
002    /**
003     * Command word class. This class should be used as a super class for
004     * all command words. There are two kinds of commands. The first type does not need
005     * a target. These commands only need to override the executeCommand(String target) method.
006     * The second type does need a target and then the executeCommand(String target), mustHaveTarget()
007     * and getTargets() must be overriden. Here comes an example:
008    <pre>
009    public class CommandMyCommand extends CommandWord {
010        public CommandMyCommand(String commandName) {
011            super(commandName);
012        }
013        &#0064;Override
014        public String[] getTargets(Game game) {
015            return new String[0]; //Should be a list of possible targets
016        }
017        &#0064;Override
018        public void executeCommand(Game game, String target) {
019            //Something that should happen when this command word is used
020        }
021    }
022    </pre>
023     */
024    public abstract class CommandWord
025    {
026        // instance variables - replace the example below with your own
027        private String command;
028    
029        /**
030         * Constructor for objects of class CommandWord
031         *
032         * @param command The command.
033         */
034        public CommandWord(String command)
035        {
036            if (!command.matches("[\\wåäö]+")) {
037                throw new IllegalArgumentException("A command word can only use [a-ö,0-9]");
038            }
039            this.command = command;
040        }
041        
042        /**
043         * Gets the command string.
044         *
045         * @return The command string.
046         */
047        public String getCommand() {
048            return command;
049        }
050        
051        /**
052         * Should be overriden. It's here the command gets executed.
053         *
054         * @param game The game.
055         * @param target The target.
056         */
057        public abstract void executeCommand(Game game, String target);
058        
059        /**
060         * Returns an array with all the possible targets at the current time.
061         * This method should be overriden if the command needs a target.
062         *
063         * @return An array wih all the current possible targets for this command.
064         */
065        public String[] getTargets(Game game) {
066            return new String[0];
067        }
068        
069        /**
070         * Should be overriden to say if this command needs a target to work.
071         *
072         * @return If this command must have a target.
073         */
074        public boolean mustHaveTarget() {
075            return false;
076        }
077    }
