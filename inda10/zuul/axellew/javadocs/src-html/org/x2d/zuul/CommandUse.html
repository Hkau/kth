001    package org.x2d.zuul;
002    import java.util.*;
003    /**
004     * Command word use. This command is used when the player is trying to use an item.
005     */
006    public class CommandUse extends CommandWord {
007        public CommandUse(String commandName) {
008            super(commandName);
009        }
010    
011        @Override
012        public String[] getTargets(Game game) {
013            Collection<Items> pItems = game.getPlayer().getItems();
014            Collection<Items> rItems = game.getPlayer().getCurrentRoom().getItems();
015            ArrayList<String> items = new ArrayList<String>();
016            checkUsable(items, pItems);
017            checkUsable(items, rItems);
018            return items.toArray(new String[0]);
019        }
020        
021        //getTarget help method.
022        private void checkUsable(ArrayList<String> items, Collection<Items> itemCollection) {
023            for (Items i : itemCollection) {
024                        if (i.getItem().isUsable()) {
025                            items.add(i.getItem().getName());
026                        }
027            }
028        }
029        /**
030         * @param target The name of the item which the player should try to use.
031         */
032        @Override
033        public void executeCommand(Game game, String target) {
034            Items item = Items.getItem(target);
035            if (game.getPlayer().getCurrentRoom().hasItem(item)) {
036                if (game.getPlayer().hasItem(item)) {
037                    System.out.println(String.format("Det finns inget föremål med namnet '%s' i det här rummet eller i din ryggsäck.", target));
038                    return;
039                }
040            }
041            if (!item.getItem().isUsable()) {
042                System.out.println("Det går inte att använda "+target);
043                return;
044            }
045            item.getItem().use(game);
046            
047        }
048        @Override
049        public boolean mustHaveTarget() {
050            return true;
051        }
052    
053    }
