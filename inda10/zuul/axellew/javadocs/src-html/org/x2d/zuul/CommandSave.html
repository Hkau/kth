001    package org.x2d.zuul;
002    import java.io.*;
003    /**
004     * Command word save. Saves the current game.
005     */
006    public class CommandSave extends CommandWord {
007        private File saveGameDir;
008        /**
009         * Constructor.
010         * @param saveDir Sets the dir where save games should be saved.
011         */
012        public CommandSave(String commandName, File saveDir) {
013            super(commandName);
014            saveGameDir = saveDir;
015        }
016        public CommandSave(String commandName) {
017            this(commandName, new File("saves/"));
018        }
019        
020        @Override
021        public String[] getTargets(Game game) {
022            return new String[]{"Ett namn du vill använda för att spara ditt spel."};
023        }
024        
025        /**
026         * @param target The name of the file which the current game should be saved to.
027         */
028        @Override
029        public void executeCommand(Game game, String target) {
030            target = target.replaceAll("([^\\w])", ""); //Makes sure only [a-ö,0-9] are used.
031            if (target==null || target == "") {
032                System.out.println("Du måste välja ett namn på filen du vill spara.");
033                return;
034            }
035            //If the save game folder does not exist create the needed folders.
036            if (!saveGameDir.exists()) {
037                    saveGameDir.mkdirs();
038            }
039            game.saveGame(new File(saveGameDir, target+SaveGameFilter.SAVE_GAME_EXTENSION));
040            System.out.println(String.format("Ditt spel har blivit sparat som '%s'.", target));
041        }
042        @Override
043        public boolean mustHaveTarget() {
044            return true;
045        }
046    }
