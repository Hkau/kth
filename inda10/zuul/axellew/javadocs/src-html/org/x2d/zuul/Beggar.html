001    package org.x2d.zuul;
002    /**
003     * A beggar NPC. He wants food and will give the player a key to the south door
004     * when he gets the food.
005     */
006    public class Beggar extends Character {
007        private boolean gotFood = false;
008        private int tCounter1 = 0;
009        private int tCounter2 = 0;
010        /**
011         * Creates a new beggar NPC.
012         */
013        public Beggar() {
014            super(Game.generateName(), "Kan du skänka mat till en hungrig stackare?");
015        }
016        
017        @Override
018        public void talk(Game g, Player p) {
019            //When the player has given him food
020            if (gotFood) {
021                String[] texts = {
022                    "Underbart med mat!",
023                    "Nom, nom, nom!",
024                };
025                System.out.println(texts[tCounter1]);
026                tCounter1 = (tCounter1 + 1) % texts.length;
027                return;
028            }
029            //If the player has no food reply some text
030            if (!p.hasItem(Items.FOOD)) {
031                String[] texts = {
032                    "Kan du skänka mat till en hungrig stackare?",
033                    "Jag är så hungrig.",
034                    "Snälla kan jag få lite mat?",
035                    "Ingen tänker på stackars lilla mig."
036                };
037                System.out.println(texts[tCounter1]);
038                tCounter1 = (tCounter1 + 1) % texts.length;
039            //If the player has food give him the key and remove the food
040            } else {
041                System.out.println("Å, mat till mig? Jag har inte ätit på en vecka!");
042                System.out.println("Jag ska hjälpa dig att öppna den södra dörren.");
043                System.out.println("Här får du nyckeln, jag hitta den på marken här utanför.");
044                p.removeItem(Items.FOOD);
045                p.addItem(Items.STORE_ROOM_KEY);
046                gotFood = true;
047            }
048        }
049    }
