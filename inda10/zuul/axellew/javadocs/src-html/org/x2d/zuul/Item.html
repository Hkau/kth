001    package org.x2d.zuul;
002    import java.io.*;
003    /**
004     * Abstract class representing an item. Methods should be overridden
005     * if the new class should do scriptet stuff. As an example:
006    <pre>
007    new Item("some name", "a description of the item") {
008        &#0064;Override
009        public boolean isUsable() {
010            return true;
011        }
012        &#0064;Override
013        public void use(Game g) {
014            //Something that should happen when this item is used.
015        }
016    };
017    </pre>
018     */
019    public abstract class Item implements Serializable
020    {
021        private String name;
022        private String description;
023        private String text;
024        private boolean isTakable;
025        
026        /**
027         * Constructor for objects of class Item
028         * @param name The name of the item.
029         * @param description A short description of the item.
030         */
031        public Item(String name, String description) {
032            this(name, description, null, false);
033        }
034        
035        /**
036         * Constructor for objects of class Item
037         * @param name The name of the item.
038         * @param description A short description of the item.
039         * @param text A text that should be used when the item is read.
040         * @param isTakable Should be <code>true</code> if the item could be put into the backpack else <code>false</code>
041         */
042        public Item(String name, String description, String text, boolean isTakable) {
043            this.name = name;
044            this.description = description;
045            this.text = text;
046            this.isTakable = isTakable;
047        }
048        
049        /**
050         * Checks if it's possible to read from this item
051         * @return <code>true</code> if there is text on the item else <code>false</code>
052         */
053        public boolean isReadable() {
054            return (getText()!=null);
055        }
056        
057        /** 
058         * Gets this item's description
059         *
060         * @return The description.
061         */
062        public String getDescription() {
063            return description;
064        }
065        
066        /** 
067         * Gets this item's name
068         *
069         * @return The name.
070         */
071        public String getName() {
072            return name;
073        }
074        
075        /**
076         * Checks if it's possible to pickup this item
077         * @return <code>true</code> if there is text on the item else <code>false</code>
078         */
079        public boolean isTakable() {
080            return isTakable;
081        }
082    
083        /**
084         * Gets the text from this item
085         * @return The text or null if there is none.
086         */
087        public String getText() {
088            return text;
089        }
090        
091        /**
092         * This method should be called when the item is used.
093         *
094         * @param g The game which this item should interact with
095         */
096        public abstract void use(Game g);
097        
098        /**
099         * Checks if it's possible to use this item. This should return true if there is an use method.
100         * @return <code>true</code> if it is possible to use this item else <code>false</code>
101         */
102        public abstract boolean isUsable();
103    }
