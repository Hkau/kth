001    package org.x2d.zuul;
002    import java.io.File;
003    import java.io.FilenameFilter;
004    /**
005     * Class used as a filter for zuul save game files.
006     */
007    public class SaveGameFilter implements FilenameFilter {
008        /**
009         * ".zul"
010         */
011        public static final String SAVE_GAME_EXTENSION = ".zul"; 
012        /**
013         * Creates a new filter that only matches files ending with ".zul".
014         */
015        public SaveGameFilter() {
016        }
017        /**
018         * Accepts only .zul files
019         *
020         * @return <code>true</code> if the file ends with .zul else <code>false</code>
021         */
022        public boolean accept(File directory, String filename) {
023            return filename.endsWith(SAVE_GAME_EXTENSION);
024        }
025    }
