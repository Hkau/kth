001    package org.x2d.zuul;
002    import java.util.*;
003    /**
004     * Command word talk. This command is used when the player is trying to talk to a character.
005     */
006    public class CommandTalk extends CommandWord {
007        public CommandTalk(String commandName) {
008            super(commandName);
009        }
010        
011        @Override
012        public String[] getTargets(Game game) {
013            Collection<Character> rChars = game.getPlayer().getCurrentRoom().getCharacters();
014            ArrayList<String> chars = new ArrayList<String>();
015            for (Character cha : rChars) {
016                chars.add(cha.getName());
017            }
018            return chars.toArray(new String[0]);
019        }
020        
021        /**
022         * @param target The name of the character which the player should try to talk to.
023         */
024        @Override
025        public void executeCommand(Game game, String target) {
026            Character cha = game.getPlayer().getCurrentRoom().getCharacter(target);
027            if (cha == null) {
028                System.out.println(String.format("Det finns ingen med namnet '%s' i det här rummet.", target));
029                return;
030            }
031            System.out.println(String.format("Du börjar prata med %s.", cha.getName()));
032            cha.talk(game, game.getPlayer());
033        }
034        @Override
035        public boolean mustHaveTarget() {
036            return true;
037        }
038    
039    }
