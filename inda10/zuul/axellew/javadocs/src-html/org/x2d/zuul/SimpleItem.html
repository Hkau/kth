001    package org.x2d.zuul;
002    /**
003     * Version of Item does not require an use method.
004     */
005    public class SimpleItem extends Item {
006        /**
007         * Constructor for objects of class Item
008         * @param name The name of the item.
009         * @param description The item's description
010         */
011        public SimpleItem(String name, String description) {
012            super(name, description);
013        }
014        
015        /**
016         * Constructor for objects of class Item
017         * @param name The name of the item.
018         * @param description The item's description
019         * @param text If there should be some text on the item that can
020         * be read. If set to <code>null</code> there is nothing to read on the item.
021         * @param isTakable Should be <code>true</code> if it is possible to take the
022         * item else it should be <code>false</code>
023         */
024        public SimpleItem(String name, String description, String text, boolean isTakable) {
025            super(name, description, text, isTakable);
026        }
027        
028        @Override
029        public void use(Game g) {
030            //Can't be used
031        }
032        
033        @Override
034        public boolean isUsable() {
035            return false;
036        }
037    
038    }
