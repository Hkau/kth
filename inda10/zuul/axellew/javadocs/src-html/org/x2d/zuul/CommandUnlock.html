001    package org.x2d.zuul;
002    import java.util.*;
003    /**
004     * Command word unlock. This command is used when the player is trying to unlock a door.
005     */
006    public class CommandUnlock extends CommandWord {
007        public CommandUnlock(String commandName) {
008            super(commandName);
009        }
010        
011       @Override
012        public String[] getTargets(Game game) {
013            Room cRoom = game.getPlayer().getCurrentRoom();
014            String[] exits = cRoom.getExits();
015            ArrayList<String> tmpArray = new ArrayList<String>();
016            for (String exit : exits) {
017                Door door = cRoom.getDoor(exit);
018                Items unlockItem = door.getUnlockItem();
019                if (door.isLocked()&&unlockItem!=null&&game.getPlayer().hasItem(unlockItem)) {
020                    tmpArray.add(exit);
021                }
022            }
023            return tmpArray.toArray(new String[0]);
024        }
025        
026        /**
027         * @param target The name of the direction which the player should try to unlock a door in.
028         */
029        @Override
030        public void executeCommand(Game game, String target) {
031            Door door = game.getPlayer().getCurrentRoom().getDoor(target);
032            if (door==null) {
033                System.out.println("Det finns inget att låsa upp åt "+target);
034                return;
035            }
036            System.out.println(String.format("Dörren i %s har blivit upplåst", target));
037            door.setLocked(false);
038        }
039        @Override
040        public boolean mustHaveTarget() {
041            return true;
042        }
043    
044    }
