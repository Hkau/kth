001    package org.x2d.zuul;
002    import java.util.*;
003    import java.io.*;
004    /**
005     * Class representing the player. The player has a backpack
006     * which stores item.
007     */
008    public class Player implements Serializable {
009        private String name;
010        private HashSet<Items> items = new HashSet<Items>(10);
011        private Room currentRoom;
012        
013        /**
014         * Constructor for objects of class Player
015         * @param name The name of the player.
016         */
017        public Player(String name) {
018            this.name = name;
019        }
020        
021        /**
022         * Gets the name of the player
023         *
024         * @return The player's name
025         */
026        public String getName() {
027            return name;
028        }
029        
030        /**
031         * Sets the player's name
032         *
033         * @param name The player's new name.
034         */
035        public void setName(String name) {
036            this.name = name;
037        }
038        
039        /**
040         * Adds an item to the players backpack
041         *
042         * @param item The item to add.
043         */
044        public void addItem(Items item) {
045            items.add(item);
046            System.out.println(String.format("Du tar imot %s och lägger i din ryggsäck.", item.getItem().getName()));
047        }
048        
049        /**
050         * Removes an item from the players backpack
051         *
052         * @param item The item to remove.
053         */
054        public void removeItem(Items item) {
055            items.remove(item);
056            System.out.println(String.format("Du plockar upp %s ur din räcksäck och använder.", item.getItem().getName()));
057        }
058        
059        /**
060         * Gets an item from the players backpack
061         *
062         * @param item The item
063         * @return <code>true</code> if found else <code>false</code>
064         */
065        public boolean hasItem(Items item) {
066            return items.contains(item);
067        }
068        
069        /**
070         * Gets a collection of all the items in the player's backpack
071         *
072         * @return A collection of all the items
073         */
074        public Collection<Items> getItems() {
075            return items;
076        }
077        
078        /**
079         * Gets the room current which the player currently is in.
080         *
081         * @return The current room.
082         */
083        public Room getCurrentRoom() {
084            return currentRoom;
085        }
086        
087        /**
088         * Sets the room current which the player currently is in.
089         *
090         * @param cRoom The the room which the player should now be in.
091         */
092        public void setCurrentRoom(Room cRoom) {
093            currentRoom = cRoom;
094            System.out.println();
095            System.out.println(currentRoom.getLongDescription());
096        }
097        /** 
098         * Try to go to one direction. If there is an exit, enter the new
099         * room, otherwise print an error message.
100         */
101        public void goRoom(String direction) 
102        {
103            // Try to leave current room.
104            Door door = getCurrentRoom().getDoor(direction);
105            if (door.isLocked()) {
106                Items unlockItem = door.getUnlockItem();
107                if (unlockItem!=null) {
108                    System.out.println(String.format("Dörren är låst och du behöver: %s för att låsa upp dörren..", unlockItem.getItem().getName()));
109                } else {
110                    System.out.println("Dörren är låst.");
111                }
112                return;
113            }
114            Room nextRoom = door.getExit(getCurrentRoom());
115            setCurrentRoom(nextRoom);
116    
117        }
118    }
