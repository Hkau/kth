001    package org.x2d.zuul;
002    import java.util.*;
003    /**
004     * Command word read. Used to read items which isReadable() returns true.
005     * The target paramenter should be the name of an item either in the
006     * player's backpack or in the room.
007     */
008    public class CommandRead extends CommandWord {
009        public CommandRead(String commandName) {
010            super(commandName);
011        }
012        
013        @Override
014        public String[] getTargets(Game game) {
015            //Checks both the player's backpack and the current room.
016            Collection<Items> pItems = game.getPlayer().getItems();
017            Collection<Items> rItems = game.getPlayer().getCurrentRoom().getItems();
018            ArrayList<String> items = new ArrayList<String>();
019                    checkReadable(items, pItems);
020                    checkReadable(items, rItems);
021            return items.toArray(new String[0]);
022        }
023        
024        //getTarget help method.
025        private void checkReadable(ArrayList<String> items, Collection<Items> itemCollection) {
026            for (Items i : itemCollection) {
027                        if (i.getItem().isReadable()) {
028                            items.add(i.getItem().getName());
029                        }
030            }
031        }
032        
033        /**
034         * @param target The name of the item which the player should try to read from.
035         */
036        @Override
037        public void executeCommand(Game game, String target) {
038            Items item = Items.getItem(target);        
039            if (!game.getPlayer().hasItem(item)) {
040                if (!game.getPlayer().getCurrentRoom().hasItem(item)) {
041                    System.out.println(String.format("Det finns inget föremål med namnet '%s' i det här rummet eller i din ryggsäck.", target));
042                    return;
043                }
044            }
045            if (!item.getItem().isReadable()) {
046                System.out.println("Det finns inget att läsa på: "+target);
047                return;
048            }
049            System.out.println("Följande står skrivet:");
050            System.out.println(item.getItem().getText());
051            
052        }
053        @Override
054        public boolean mustHaveTarget() {
055            return true;
056        }
057    
058    }
