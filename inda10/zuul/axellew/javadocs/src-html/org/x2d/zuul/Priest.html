001    package org.x2d.zuul;
002    /**
003     * The priest NPC that has lost his prayer beads
004     * and when the player gives them to the priest
005     * he gives player a seal which will let him get into
006     * the temple.
007     */
008    public class Priest extends Character {
009        private Item seal;
010        boolean gotPrayerBeads = false;
011        private int tCounter1 = 0;
012        private int tCounter2 = 0;
013        public Priest() {
014            super(Game.generateName(), "Argh!");
015        }
016        
017        @Override public void talk(Game g, Player p) {
018            //Replies when the player has given the Priest the prayer beads
019            if (gotPrayerBeads) {
020                String[] texts = {
021                    "Gud är stor!",
022                    "Jag hoppas att du får all välgång!",
023                };
024                System.out.println(texts[tCounter1]);
025                tCounter1 = (tCounter1 + 1) % texts.length;
026                return;
027            }
028            //Some replies to give the player a clue of what the priest want for an item
029            if (!p.hasItem(Items.PRAYER_BEADS)) {
030                String[] texts = {
031                    "Jag förstår inte vart mitt böneband har tagit vägen!",
032                    "Du kanske kan hjälpa mig att hitta den?",
033                    "Jag vet att jag hade den när jag var och åt.",
034                    "Jag var inne i lagerrummet sen."
035                };
036                System.out.println(texts[tCounter1]);
037                tCounter1 = (tCounter1 + 1) % texts.length;
038            //If the player has the prayer beads give it to the priest and the player gets
039            //a seal which will help him get into the temple.
040            } else {
041                System.out.println("Ah du hittade den!");
042                p.removeItem(Items.PRAYER_BEADS);
043                System.out.println("Aha, du vill komma ut ur borgen.");
044                System.out.println("Jag tror du kan hitta det du letar efter i templet.");
045                System.out.println("Här får du ett mitt sigill som du kan visa för vakten så att du kommer in.");
046                p.addItem(Items.SEAL);
047                gotPrayerBeads = true;
048            }
049        }
050    }
